      <p><strong>xacro</strong> (XML Macros) makes URDF more maintainable:</p>
      
      <pre>{`
<?xml version="1.0"?>
<robot xmlns:xacro="http://www.ros.org/wiki/xacro" name="humanoid">

  <!-- Define macros -->
  <xacro:macro name="leg" params="side reflect">
    <link name="upper_leg_\${side}">
      <visual>
        <geometry>
          <cylinder radius="0.06" length="0.4"/>
        </geometry>
      </visual>
    </link>

    <joint name="hip_\${side}" type="revolute">
      <parent link="pelvis"/>
      <child link="upper_leg_\${side}"/>
      <origin xyz="0 \${reflect*0.1} 0" rpy="0 0 0"/>
      <axis xyz="0 1 0"/>
      <limit lower="-1.57" upper="1.57" effort="150" velocity="2.0"/>
    </joint>
  </xacro:macro>

  <!-- Use macros -->
  <xacro:leg side="right" reflect="-1"/>
  <xacro:leg side="left" reflect="1"/>

</robot>
      `}</pre>
      
      <p>Convert xacro to URDF:</p>
      
      <pre>{`
xacro robot.xacro > robot.urdf
      `}</pre>
